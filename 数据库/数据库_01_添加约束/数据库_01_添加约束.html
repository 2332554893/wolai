1.创建数据库

--创建 test_1 数据库
create database test_1

2.在数据库中创建数据表

use test_1
--创建数据表并设置类型
create table test
(
  t1 int identity not null,
  t2 int,
  t3 int,
  t4 int,
  t5 int 
)

3.向表中添加三种约束，并删除约束

--向表中t1添加主键约束
alter table test
add constraint PK_t1 primary key(t1)

--向t5添加check约束 取值 0-100 之间
alter table test
add constraint CK_t5 check(t5 between 0 and 100)

--向t4添加一个默认值约束 默认为1
alter table test
add constraint DF_t4 default(1) for t4

--删除上面添加的约束
alter table test
drop constraint PK_t1,CK_t5,DF_t4

4.修改表结构

--修改表结构
-- alter table 表名

-- add constraint 约束名 约束 --添加约束
add constraint CK_q check(BorrowedQuantity between 0 and 5)

-- alter column 列名 数据类型 --修改列的数据类型
alter column Name varchar(20)

-- add 字段名 数据类型  --添加列
add  Defaults  int

-- drop constraint 约束名 -删除约束
drop constraint CK_q 

-- drop column 列名  --删除列
drop column Defaults

-- drop table 表名  --删除表
drop table rea
